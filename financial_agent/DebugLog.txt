### 1.3 重構 `run_sql_query(args: dict, file_registry: dict) -> str`
此函數必須使用 `pyodbc` 連接真實資料庫，實作 Big5 編碼防禦，執行 `pd.read_sql`，並將完整大數據安全儲存為 `utf-8-sig` 的 CSV 檔，同時在回傳末端埋入特殊隔離標記。請完整實現以下內容：
```python
def run_sql_query(args: dict, file_registry: dict) -> str:
    """執行模型生成的 SELECT SQL（唯讀），利用 pyodbc 與編碼設定連線 Teradata，
    並將查詢結果存為 CSV，供 Coder Agent 進行互動式多維度樞紐分析。"""
    sql = args.get("sql", "")
    print(f"📊 [Teradata SQL] {sql[:160]}")
    
    # 填入您測試成功的企業環境資料
    driver_name = "Teradata Database ODBC Driver 17.20"
    server_name = "aaa"
    username = "aaa" 
    password = "aaa@aaa#aaa"

    conn_str = (
        f"DRIVER={{{driver_name}}};"
        f"DBCNAME={server_name};"
        f"UID={username};"
        "PWD={" + password + "};" +
        f"Authentication=SYSTEM;" 
        f"TMODE=TERA;"
    )
    
    try:
        # 1. 建立 Teradata 連線
        conn = pyodbc.connect(conn_str, autocommit=True)
        
        # 2. 嵌入編碼修正防禦機制，防範亂碼與 "not type" 錯誤
        try:
            conn.setdecoding(pyodbc.SQL_CHAR, encoding="big5")
            conn.setdecoding(True, encoding="big5")
        except Exception:
            pass  # 驅動程式不支援時跳過，確保基本連線不受影響
            
        # 3. 使用 Pandas 直接讀取 SQL
        df = pd.read_sql(sql, conn)
        conn.close()

        if df.empty:
            return "✅ Teradata 查詢成功執行，但該條件下資料庫無符合的紀錄。"
            
        # 4. 數據隔離核心：將完整數據集落檔至本地快取，防止 Token 爆炸與 AI 幻覺
        # 使用 utf-8-sig 編碼確保 Excel 直接點開不亂碼，且支援中文
        os.makedirs(RUNTIME.cache_dir, exist_ok=True)
        csv_path = os.path.join(RUNTIME.cache_dir, "db_query_result.csv")
        df.to_csv(csv_path, index=False, encoding='utf-8-sig')
        
        # 5. 只擷取前 30 筆的 Markdown 預覽餵回給總結模型
        preview_df = df.head(30)
        
        result_str = (
            f"✅ Teradata 查詢成功！共從 `HDATA.SAL_DRAFT_RPT_HC` 撈取 {len(df)} 筆明細資料。

"
            f"【資料庫結果預覽 (前 30 筆)】
{preview_df.to_markdown()}

"
            f"💡 [系統管線指令]：完整大數據集已安全暫存於本地快取：'{csv_path}'。
"
            f"後續步驟若需利用 `run_python_code` 產生多維度樞紐分析表(Pivot Table)或繪製圖表，"
            f"請直接在任務描述中命令 Coder 讀取該檔案進行分析：
"
            f"```python
import pandas as pd
import os
df = pd.read_csv(r'{csv_path}')
```
"
            f"嚴禁模型自行硬編碼(Hardcode)或虛構數據。

"
            f"__CSV_CACHE_PATH__:{csv_path}"  # 🚀 埋入特殊管線標記，供外層攔截
        )
        return result_str
        
    except Exception as e:
        return f"❌ Teradata 資料庫 SQL 執行失敗，錯誤訊息：
{e}"
```

### 1.4 修改 `_execute_tool_loop` 函數
請修改 `_execute_tool_loop`，在 `gather` 收集完證據後，攔截特殊標記 `__CSV_CACHE_PATH__` 並將路徑賦值給 `resp.csv_cache_path`：
```python
def _execute_tool_loop(plan: PlanningResult, user_prompt: str,
                       file_registry: dict, resp: AgentResponse) -> None:
    _progress("📂 解析檔案、收集證據中…")
    evidence = gather(plan, user_prompt, file_registry, resp)
    
    # 🚀 新增攔截邏輯：從證據原文中抽離隱藏的實體 CSV 快取路徑
    for ev in evidence:
        if "__CSV_CACHE_PATH__:" in ev:
            resp.csv_cache_path = ev.split("__CSV_CACHE_PATH__:")[1].strip()
            break
            
    print("
  🧩 [Synthesize] 總結 agent 整合證據...")
    _progress("🧩 整合分析、撰寫報告中…（這步較久，請稍候）")
    want_viz = _wants_visualization(user_prompt, plan)
    synthesis = _synthesize(user_prompt, plan, evidence, resp, want_viz)

        _progress("📈 繪製圖表中…")
    _present_synthesis(synthesis, resp, file_registry, want_viz)

### 2.2 實作 Essbase 專屬高階展示函數
請在 `app.py` 中宣告以下兩個專屬的高階多維展示函數，充分發揮 Essbase 內建層級路徑與時間維度預處理的特徵：

```python
def _render_essbase_aggrid(csv_path: str):
    """🚀 高級功能：將 Essbase 格式資料透過 AgGrid 進行企業級樹狀群組展示"""
    if not os.path.exists(csv_path):
        st.error("❌ 找不到資料庫快取 CSV 檔案。")
        return
    
    df = pd.read_csv(csv_path)
    st.markdown(f"📊 **資料庫完整數據明細 (共 {len(df)} 筆紀錄)**")
    
    if not HAS_ADVANCED_GRID:
        st.dataframe(df, use_container_width=True)
        st.caption("💡 提示：安裝 `streamlit-aggrid` 可解鎖高級多層級群組展開功能。")
        return

    # 建立企業級 AgGrid 屬性設定
    gob = GridOptionsBuilder.from_dataframe(df)
    gob.configure_pagination(paginationAutoPageSize=False, paginationPageSize=15) # 分頁限制
    gob.configure_side_bar() # 啟用高級多功能篩選側邊欄
    gob.configure_default_column(groupable=True, value=True, enableRowGroup=True, filter=True, sortable=True)
    
    # ⚡ 針對 Essbase 組織特徵，預設啟用高階群組樹狀優勢
    if "PARENT_SITE_ORG" in df.columns and "CHILD_SITE_ORG" in df.columns:
        gob.configure_column("PARENT_SITE_ORG", rowGroup=True, hide=True)
        gob.configure_column("CHILD_SITE_ORG", rowGroup=True, hide=True)
    
    gridOptions = gob.build()
    
    AgGrid(
        df,
        gridOptions=gridOptions,
        columns_auto_size_mode=ColumnsAutoSizeMode.FIT_CONTENTS,
        theme="alpine",  # 經典財務審計專業主題
        height=450
    )


def _render_essbase_pivotjs(csv_path: str, msg_id: str):
    """🚀 高級功能：將 Essbase 格式資料透過 iframe 嵌入 PivotTableJS，完美重現 Excel 自由拖拉拽體驗"""
    if not os.path.exists(csv_path):
        return
    if not HAS_ADVANCED_GRID:
        st.warning("請先於環境中安裝 `streamlit-pivottablejs`。")
        return
        
    df = pd.read_csv(csv_path)
    
    # 防止多輪對話並發操作導致實體 HTML 互相覆蓋
    html_filename = f"temp_pivot_{msg_id}.html"
    
    # 呼叫產生拖拉拽樞紐分析 UI
    pivot_ui(df, outfile=html_filename)
    
    if os.path.exists(html_filename):
        with open(html_filename, "r", encoding="utf-8") as f:
            html_content = f.read()
        
        # 透過 Streamlit HTML 元件寬度填滿嵌入網頁
        st.components.v1.html(html_content, height=600, scrolling=True)
        
        # 自動資源回收清理暫存檔
        try:
            os.remove(html_filename)
        except OSError:
            pass
```

### 2.3 重構對話渲染器 `_render_payload` (升級為 `_render_payload_with_tabs`)
請將原本單一文字流的 `_render_payload` 函數，重構為支援多標籤隔離展示的高級版本。必須保證文字報告中清除可能殘留的後端路徑標記：
```python
def _render_payload_with_tabs(payload: dict, msg_idx: int):
    """重構訊息渲染器：採用多 Tab 結構，完美將 AI 報告與真實大數據表格進行空間隔離"""
    csv_path = payload.get("csv_cache_path", "")
    
    # 情境 A：如果有資料庫大數據快取，啟動進階 Tabs 數據隔離展現
    if csv_path and os.path.exists(csv_path):
        tab1, tab2, tab3 = st.tabs(["🤖 AI 智能解讀報告", "🗂️ 企業級數據網格 (AgGrid)", "🔀 自由拖拉樞紐分析 (Excel UI)"])
        
        with tab1:
            if payload.get("content"):
                # 清除可能殘留的後端隱藏路徑標記，維持終端用戶畫面純淨
                clean_text = payload["content"].split("__CSV_CACHE_PATH__:")[0].strip()
                st.markdown(clean_text)
            # 渲染其他靜態圖表
            if payload.get("plotly_jsons"):
                _render_plotly_jsons(payload["plotly_jsons"])
            for fig in payload.get("figures", []) or []:
                st.plotly_chart(fig, use_container_width=True)
            for html in payload.get("tables", []) or []:
                st.html(html)
            for img in payload.get("images", []) or []:
                if os.path.exists(img): st.image(img)
                
        with tab2:
            _render_essbase_aggrid(csv_path)
            
        with tab3:
            _render_essbase_pivotjs(csv_path, f"{msg_idx}_{round(os.path.getmtime(csv_path))}")
            
    # 情境 B：一般文件庫 RAG 回答或閒聊問答，維持直覺的單一文字流
    else:
        if payload.get("content"):
            st.markdown(payload["content"])
        if payload.get("plotly_jsons"):
            _render_plotly_jsons(payload["plotly_jsons"])
        for fig in payload.get("figures", []) or []:
            st.plotly_chart(fig, use_container_width=True)
        for html in payload.get("tables", []) or []:
            st.html(html)
        for img in payload.get("images", []) or []:
            if os.path.exists(img): st.image(img)
```
